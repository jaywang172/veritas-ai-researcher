# Veritas v3.0 Environment Configuration Template
# 複製此文件為 .env 並填入您的 API Keys

# OpenAI API Key (必須)
# 獲取方式：https://platform.openai.com/account/api-keys
OPENAI_API_KEY=your_openai_api_key_here

# Tavily Search API Key (可選，用於網路搜尋)  
# 獲取方式：https://tavily.com/
TAVILY_API_KEY=your_tavily_api_key_here

# Google API Key (可選，如果使用 Gemini 模型)
# 獲取方式：https://makersuite.google.com/app/apikey
GOOGLE_API_KEY=your_google_api_key_here

# LangChain 追蹤 (可選，用於除錯)
# LANGCHAIN_TRACING_V2=true
# LANGCHAIN_API_KEY=your_langchain_api_key_here
